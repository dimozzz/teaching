\begin{definition*}
    \deftext{Матрица Уолша--Адамара} $H_k$~--- это вещественная матрица размера $2^k \times 2^k$, которую
    можно определить индуктивно: $H_0 \coloneqq 1$,
    $$
        H_{k + 1} \coloneqq \left(
            \begin{array}{cc}
              H_k & H_k \\
                H_k & -H_k
            \end{array}
        \right).
    $$
\end{definition*}
