\libdefinition{real-closed-field}

\task{
    Выведите в теории линейно упорядоченных полей:
    $$
        x < y \longrightarrow \exists u\, x < u < y.
    $$
}


\begin{proposition}
    Для любого линейно упорядоченного поля $\mathbb{F}$ следующие условия эквивалентны:
    \begin{enumcyr}
        \item $\mathbb{F}$ вещественно замкнуто;
        \item $\mathbb{F}$ как поле не является алгебраически замкнутым, но его расширение
            $\mathbb{F}[\sqrt{-1}]$ алгебраически замкнуто.
    \end{enumcyr}
\end{proposition}

\task{
    Покажите, что для любого линейно упорядоченного поля $\mathbb{F}$ следующие условия эквивалентны:
    \begin{enumcyr}
        \item $\mathbb{F}$ вещественно замкнуто;
        \item в $\mathbb{F}$ истинно $\forall x\, (0 < x \rightarrow \exists u\, x = u \cdot u)$, а
            также все $\sigma$-формулы вида:
            $$
                \forall \overline{u}\, \forall x\, \forall y\,
                (\mathrm{p}_n(x; \overline{u}) \cdot \mathrm{p}_n(y; \overline{u}) < 0 \rightarrow
                \exists z\, (x < z < y \wedge \mathrm{p}_n(z; \overline{u}) = 0)),
            $$
            т.е.\ для $\mathbb{F}$ верна <<полиномиальная версия>> теоремы о среднем значении.
    \end{enumcyr}
}
