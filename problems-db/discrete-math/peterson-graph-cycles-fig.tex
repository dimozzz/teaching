    \def\R{2.1}
    \def\r{1}

        \node[g-vert] (a\a) at (90 - \a * 360 / 5:\R) {};

        \draw[thick] (a\a) -- (b\a);

    }

            \draw[thick] (b\a) -- (b\j);
    }
