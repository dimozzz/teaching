Для функции $f\colon \{\pm 1\}^n \to \{\pm 1\}$ докажите, что:
\begin{enumcyr}
    \item $\Var[f] = 4 \measure{dist}(f, 1) \cdot \measure{dist}(f, -1)$;
    \item $2 \varepsilon \le \Var[f] \le 4 \varepsilon$, где $\varepsilon = \min \{\measure{dist}(f, 1),
        \measure{dist}(f, -1)\}$.
\end{enumcyr}
