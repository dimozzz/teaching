Докажите, что следующие предикаты являются арифметичными:
\begin{enumcyr}
    \item $x < y$;
    \item $x = 0$;
    \item $x = 1$;
    \item $x = c$, где $c$~--- это некоторая натуральная константа;
    \item $a \bmod b = r$;
    \item $a$~--- это степень двойки;
    \item $a$~--- это степень четверки.
\end{enumcyr}
