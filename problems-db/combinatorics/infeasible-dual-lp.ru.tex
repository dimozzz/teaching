Предъявите такую несовместную задачу линейного программирования, что двойственная задача будет также
несовместна.
\tags{линейное программирование, комбинаторика}

\answer{
    $2 x_1 - x_2 \to \max$

    \begin{align*}
      x_1 - x_2 & \le 1 \\
      -x_1 + x_2 & \le -2 \\
      x_1, x_2 & \ge 0
    \end{align*}
}
