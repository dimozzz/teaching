Найдите решение следующей задачи линейного программирования:
$\max x - y$, при условии
\begin{align*}
  3x + y & \le 24 \\
  -2x + y & \le -1 \\
  x - 2y & \le 1 \\
  x, y & \ge 0
\end{align*}
Сформулируйте задачу, двойственную данной.

\answer{
    Точка $(7, 3)$, для неё $3x + y = 24$, $-2x + y = -11$, $x - 2y = 1$, максимум $4$.

    \begin{tikzpicture}[black]
        \begin{axis}[
            width = 7cm,
            height = 7cm,
            xmin = -1.5,
            xmax = 12.5,
            ymin = -1.5,
            ymax = 12.5,
            axis x line = middle,
            axis y line = middle,
            axis line style = {thick},
            xtick = {-1, 0, ..., 12},
            ytick = {-1, 0, ..., 12},
            xticklabels = {,,},
            yticklabels = {,,},
            grid = both]

            \path[name path = h] (-1, 0) -- (10, 0);
            \path[name path = v] (0, -1) -- (0, 10);
            \addplot[draw = none, name path = a, domain = -1:8, samples = 100] {4 + x};
            \addplot[draw = none, name path = b, domain = -1:8, samples = 100] {(2 * x - 5) / 3};
            \addplot[draw = none, name path = c, domain = -1:8, samples = 100] {24 - 3 * x};

            \path[name intersections = {of = b and h, by = i1}];
            \path[name intersections = {of = b and c, by = i2}];
            \path[name intersections = {of = a and c, by = i3}];
            \path[name intersections = {of = a and v, by = i4}];

            \fill[green!80!black, opacity = 0.2] (0, 0) -- (i1) -- (i2) -- (i3) -- (i4) -- cycle;

            \addplot[thick, domain = -1:10, samples = 100, color = blue] {4 + x};
            \addplot[thick, domain = -1:9.2, samples = 100, color = blue] {(2 * x - 5) / 3};
            \addplot[thick, domain = -1:10, samples = 100, color = blue] {24 - 3 * x};

            \addplot[domain = -1:8.1, samples = 100, color = red] {x - 4};

            \node[fill = white, rounded corners = 3pt, inner sep = 1pt] at (9.5, 10.5)
                {\scriptsize $-x + 4 = 4$};
            \node[fill =  white, rounded corners = 3pt, inner sep = 1pt] at (10, 5.1)
                {\scriptsize $2x - 3y = 5$};
            \node[fill = white, rounded corners = 3pt, inner sep = 1pt] at (10, 1.7)
                {\scriptsize $3x + y = 24$};

            \addplot[->, dashed, domain = 2:7, samples = 100, color = red] {-x + 10};
            \node[fill = white, rounded corners = 3pt, inner sep = 1pt] at (2, 8.5)
                {\scriptsize \textcolor{red}{$x - y$}};
	\end{axis}
\end{tikzpicture}
}
